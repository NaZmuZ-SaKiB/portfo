002 : Password checker

It is a password checker program. It will tell you how many times your password was hacked. I have used 'requests', 'hashlib' and 'sys' module in the code.

"Make sure your password is safe."

Running the program and cheaking passwords

You can also enter multiple passwords at a time

Here is the code


You can run it in any Interpreter


import requests
import hashlib
import sys

print('******Welcome to password checker******')
print('Please wait......')

def request_api_data(query_char):
url = 'https://api.pwnedpasswords.com/range/'+query_char
res = requests.get(url)
if res.status_code != 200:
raise RuntimeError(f'Error fetching: {res.status_code}, check the API and try again')
return res

def get_pass_leak_count(hashes, hash_to_check):
hashes = (line.split(':') for line in hashes.text.splitlines())
for h, count in hashes:
if h == hash_to_check:
return count
return 0

def pwned_api_check(password):
sha1pass = hashlib.sha1(password.encode('utf-8')).hexdigest().upper()
first_5_pass, rest_pass = sha1pass[:5], sha1pass[5:]
response = request_api_data(first_5_pass)
return get_pass_leak_count(response, rest_pass)

def main(args):
for password in args:
hack_count = pwned_api_check(password)
if hack_count:
print(f"Be careful........\nThe password '{password}' was hacked {hack_count} times\n")
else:
print(f"Great.......\nThe password '{password}' is a strong password\n")
main(sys.argv[1:])
